```python
# W pliku utils/lesson_progress.py lub odpowiednim pliku postƒôpu lekcji
def award_fragment_xp(lesson_id, fragment_type, xp_amount):
    """
    Przyznaj XP i Neurocoin za uko≈Ñczenie fragmentu lekcji
    
    Args:
        lesson_id: ID lekcji
        fragment_type: 'intro', 'opening_quiz', 'content', 'practical_exercises', 'closing_quiz', 'summary'
        xp_amount: Ilo≈õƒá XP do przyznania
    """
    users_data = load_user_data()
    username = st.session_state.username
    
    if username in users_data:
        user_data = users_data[username]
        
        # Struktura: lesson_progress[lesson_id][fragment_type] = {'completed': True, 'xp_awarded': 10}
        lesson_progress = user_data.get('lesson_progress', {})
        
        if lesson_id not in lesson_progress:
            lesson_progress[lesson_id] = {}
        
        # Sprawd≈∫ czy XP za ten fragment ju≈º zosta≈Ço przyznane
        fragment_key = f"{fragment_type}_xp_awarded"
        if not lesson_progress[lesson_id].get(fragment_key, False):
            # Dodaj XP
            current_xp = user_data.get('xp', 0)
            user_data['xp'] = current_xp + xp_amount
            
            # Dodaj Neurocoin r√≥wne ilo≈õci XP ‚Üê KLUCZOWA FUNKCJONALNO≈öƒÜ
            current_neurocoin = user_data.get('neurocoin', 0)
            user_data['neurocoin'] = current_neurocoin + xp_amount
            
            # Zaznacz ≈ºe XP zosta≈Ço przyznane
            lesson_progress[lesson_id][fragment_key] = True
            lesson_progress[lesson_id][f"{fragment_type}_completed"] = True
            lesson_progress[lesson_id][f"{fragment_type}_xp"] = xp_amount
            lesson_progress[lesson_id][f"{fragment_type}_neurocoin"] = xp_amount  # ‚Üê NOWE
            lesson_progress[lesson_id][f"{fragment_type}_timestamp"] = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
            
            user_data['lesson_progress'] = lesson_progress
            
            # Zapisz dane
            users_data[username] = user_data
            save_user_data(users_data)
            
            # Od≈õwie≈º session_state
            st.session_state.user_data = user_data
            
            return True, xp_amount
    
    return False, 0
```

```python
# Plik: initialize_neurocoin.py
import json
import os
from datetime import datetime

def initialize_neurocoin_for_existing_users():
    """
    Inicjalizuje Neurocoin dla wszystkich istniejƒÖcych u≈ºytkownik√≥w
    Przyznaje Neurocoin r√≥wne obecnemu XP
    """
    print("üß† Inicjalizacja systemu Neurocoin...")
    print("=" * 50)
    
    # Za≈Çaduj dane u≈ºytkownik√≥w
    users_file = 'users_data.json'
    
    if not os.path.exists(users_file):
        print("‚ùå Plik users_data.json nie zosta≈Ç znaleziony!")
        return
    
    with open(users_file, 'r', encoding='utf-8') as f:
        users_data = json.load(f)
    
    print(f"üìä Znaleziono {len(users_data)} u≈ºytkownik√≥w do aktualizacji")
    
    updated_count = 0
    
    for username, user_data in users_data.items():
        # Sprawd≈∫ czy u≈ºytkownik ju≈º ma Neurocoin
        if 'neurocoin' not in user_data:
            # Przyznaj Neurocoin r√≥wne obecnemu XP
            current_xp = user_data.get('xp', 0)
            user_data['neurocoin'] = current_xp
            
            print(f"‚úÖ {username}: {current_xp} XP ‚Üí {current_xp} Neurocoin")
            updated_count += 1
        else:
            print(f"‚è≠Ô∏è {username}: ju≈º posiada Neurocoin ({user_data.get('neurocoin', 0)})")
    
    # Zapisz zaktualizowane dane
    with open(users_file, 'w', encoding='utf-8') as f:
        json.dump(users_data, f, indent=2, ensure_ascii=False)
    
    print("\n" + "=" * 50)
    print(f"üéâ Migracja zako≈Ñczona!")
    print(f"üìà Zaktualizowano: {updated_count} u≈ºytkownik√≥w")
    print(f"üìÖ Data migracji: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
    
    return True

if __name__ == "__main__":
```
